class Ejercicio:
    """
    Representa un ejercicio de entrenamiento físico.

    :param nombre_ejercicio: Nombre del ejercicio.
    :type nombre_ejercicio: str
    :param grupo_muscular: Grupo muscular principal que trabaja el ejercicio.
    :type grupo_muscular: str
    :param nivel_dificultad: Nivel de dificultad del ejercicio (e.g., Principiante, Intermedio, Avanzado).
    :type nivel_dificultad: str
    :param repeticiones: Número de repeticiones recomendadas.
    :type repeticiones: int
    :param series: Número de series recomendadas.
    :type series: int
    """

    def __init__(self, nombre_ejercicio, grupo_muscular, nivel_dificultad, repeticiones, series):
        self.nombre_ejercicio = nombre_ejercicio
        self.grupo_muscular = grupo_muscular
        self.nivel_dificultad = nivel_dificultad
        self.repeticiones = repeticiones
        self.series = series

    def mostrar_informacion(self):
        """
        Muestra la información completa del ejercicio.
        """
        print("NOMBRE: " + self.nombre_ejercicio)
        print("GRUPO MUSCULAR: " + self.grupo_muscular)
        print("NIVEL DE DIFICULTAD: " + self.nivel_dificultad)
        print("REPETICIONES: " + str(self.repeticiones))
        print("SERIES: " + str(self.series))
        print("\n")

    def es_apto_para(self, nivel_usuario):
        """
        Evalúa si el ejercicio es apto para el nivel del usuario.

        :param nivel_usuario: Nivel de experiencia del usuario.
        :type nivel_usuario: str
        :return: Verdadero si el usuario tiene el nivel adecuado, Falso en caso contrario.
        :rtype: bool
        """
        niveles = ["Principiante", "Intermedio", "Avanzado"]
        return niveles.index(nivel_usuario) >= niveles.index(self.nivel_dificultad)

class Rutina(Ejercicio):
    """
    Representa una rutina de ejercicios, que incluye varios ejercicios individuales.

    :param nombre_rutina: Nombre de la rutina.
    :type nombre_rutina: str
    :param grupo_muscular: Grupo muscular principal al que está dirigida la rutina.
    :type grupo_muscular: str
    :param nivel_dificultad: Nivel de dificultad de la rutina.
    :type nivel_dificultad: str
    :param repeticiones: Repeticiones iniciales para la rutina (generalmente 0).
    :type repeticiones: int
    :param series: Series iniciales para la rutina (generalmente 0).
    :type series: int
    :param objetivo: Objetivo principal de la rutina, como “Ganar masa muscular”.
    :type objetivo: str
    :param nivel_rutina: Nivel de la rutina, como “Intermedio” o “Avanzado”.
    :type nivel_rutina: str
    """

    def __init__(self, nombre_rutina, grupo_muscular, nivel_dificultad, repeticiones, series, objetivo, nivel_rutina):
        super().__init__(nombre_rutina, grupo_muscular, nivel_dificultad, repeticiones, series)
        self.objetivo = objetivo
        self.nivel_rutina = nivel_rutina
        self.lista_ejercicios = []

    def remover_ejercicio(self, nombre):
        """
        Remueve un ejercicio de la rutina basado en su nombre.

        :param nombre: Nombre del ejercicio a remover.
        :type nombre: str
        """
        for ejercicio_ in self.lista_ejercicios:
            if ejercicio_.nombre_ejercicio == nombre:
                self.lista_ejercicios.remove(ejercicio_)
                break
        else:
            print("Ese ejercicio no se encuentra en la rutina.")

    def mostrar_rutina(self):
        """
        Muestra la rutina completa con todos los ejercicios y sus detalles.

        print('############################')
        print(f"     ** RUTINA: {self.nombre_ejercicio} **")

        print(f"   Nivel de la Rutina: {self.nivel_rutina}")

        print("   Ejercicios en la rutina:")
        print('============================')
        for ejercicio in self.lista_ejercicios:
            ejercicio.mostrar_informacion()
        print('############################\n')
